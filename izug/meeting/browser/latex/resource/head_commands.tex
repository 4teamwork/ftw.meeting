% checkbox command
\newcommand{\checkbox}[1]{
    \tiny
    \setlength{\tabcolsep}{3pt}
    \begin{tabular}{|p{5pt}|}
        \hline
        \tiny #1 \\
        \hline
    \end{tabular}
    \normalsize
}


% meetingitem command
\newcommand{\meetingitem}[2]{
    \rule[1ex]{18.5cm}{1pt}
    {\bf #1}
    \marginpar{
        \vspace{-4mm}
        \begin{tabular}{|p{5cm}}
            #2 \\
        \end{tabular}
    }
}
