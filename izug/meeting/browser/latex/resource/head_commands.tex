\newcounter{meetingCounter}
\setcounter{meetingCounter}{0}
\newcommand{\meeting}[4]{
    \addtocounter{meetingCounter}{1}
    \paragraph{\arabic{meetingCounter}. #1}
    #2
    \def\third{#3}%
    \def\nothing{}%
    \ifx\third\nothing
        \newline
    \else
        \\{\bf Ergebnis:}
        #3\newline
    \fi
    \begin{tabular}{p{.99\textwidth}}
        \raggedleft
        {\small\em #4}
    \end{tabular}
}
